import random

secret = random.randint(1, 100)
attempts = 0
guess = 0

while guess != secret:
    guess = int(input("נחש: "))
    attempts += 1
    if guess < secret:
        print("גבוה יותר!")
    elif guess > secret:
        print("נמוך יותר!")

print(f"🎉 ניחשת ב-{attempts} ניסיונות!")
שלב 3 — הגרסה המלאה

גרסה 3 — עם כל השיפורים

גרסה מלאה — try/except + הגבלת ניסיונות + ניגנוד
import random

def play_game():
    secret = random.randint(1, 100)
    attempts = 0
    max_attempts = 10

    print("ברוך הבא למשחק ניחוש המספר!")
    print(f"יש לך {max_attempts} ניסיונות לנחש מספר בין 1-100")

    while attempts < max_attempts:
        try:
            guess = int(input(f"ניסיון {attempts+1}/{max_attempts}: "))
        except ValueError:
            print("הכנס מספר בלבד!")
            continue

        attempts += 1

        if guess == secret:
            print(f"🎉 כל הכבוד! ניחשת ב-{attempts} ניסיונות!")
            return
        elif guess < secret:
            print("📈 גבוה יותר!")
        else:
            print("📉 נמוך יותר!")

    print(f"😢 נגמרו הניסיונות. המספר היה {secret}")

play_game()
