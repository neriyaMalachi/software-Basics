# משתנים + רשימה + לולאה + תנאי + פונקציה
def analyze_grades(grades):
    passing = []
    for g in grades:
        if g >= 60:
            passing.append(g)
    avg = sum(passing) / len(passing) if passing else 0
    return {"עברו": len(passing), "ממוצע": round(avg, 1)}

result = analyze_grades([55, 82, 91, 47, 76])
print(result)  # {'עברו': 3, 'ממוצע': 83.0}
